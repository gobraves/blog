打包
tar -czf fileFolder.tar.gz fileFolder
解包
tar -xvf fileFolder.tar.gz

.tar.gz压缩
tar zcvf fileName.tar.gz fileFolder
.tar.gz解压
tar -zxvf fileName.tar.gz

.tar.bz2压缩
tar jcvf fileName.tar.bz2 fileFolder
.tar.bz2解压
tar jxvf fileName.tar.bz2
